> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the two backend correctness bugs and add the Critical+High test pack (59 tests) and CI workflows for Aegis Company Profile v2.1, shipping in 5 PRs each with auto-merge enabled.

**Architecture:** No new modules, no new tables. PR 1 introduces a pytest + Playwright test harness with a deterministic mock-Claude bank. PRs 3–5 fill in test coverage (TC-O/V backend, TC-U/A Playwright, TC-R + CI workflows). Five tests run nightly against real Claude.

**Tech Stack:** FastAPI + SQLAlchemy 2.0 + Postgres + Celery + Redis (backend); React 18 + Vite + Zustand (frontend); pytest + pytest-asyncio + httpx + pytest-mock + Playwright (test infra); GitHub Actions (CI).

```python
# ── Mock Claude ───────────────────────────────────────────────────────────────

class MockClaudeBank:
    """Returns canned responses keyed by (agent, company, field).

    Raises AssertionError on unknown keys — forces tests to declare fixtures
    explicitly. No silent fallback (most common source of false-green LLM tests).
    """

    def __init__(self):
        self._cache: dict[str, dict] = {}
        self._calls: list[dict] = []

    def for_call(self, agent: str, company: str, field: str) -> dict:
        key = f"{agent}__{company}__{field}.json"
        self._calls.append({"agent": agent, "company": company, "field": field})
        if key in self._cache:
            return self._cache[key]
        path = CLAUDE_RESPONSES_DIR / key
        if not path.exists():
            raise AssertionError(
                f"Missing Claude fixture: {key}.\n"
                f"  Record it (LIVE_CLAUDE=1 python tests/scripts/record_fixtures.py "
                f"--agent {agent} --company {company} --field {field}) "
                f"or add a stub JSON to tests/fixtures/claude_responses/."
            )
        data = json.loads(path.read_text())
        self._cache[key] = data
        return data

    @property
    def calls(self) -> list[dict]:
        return list(self._calls)


@pytest.fixture
def mock_claude(monkeypatch) -> MockClaudeBank:
    """Patches the Anthropic SDK with the MockClaudeBank.

    Seeder/validator system prompts must include `# META: agent=X company=Y field=Z`.
    """
    bank = MockClaudeBank()
    import anthropic

    class _FakeBlock:
        def __init__(self, text: str):
            self.type = "text"
            self.text = text

    class _FakeAnthropicResponse:
        def __init__(self, payload):
            self.content = [_FakeBlock(json.dumps(payload))]
            self.stop_reason = "end_turn"

    class _FakeMessages:
        async def create(self, **kwargs):
            meta = _extract_meta(kwargs)
            resp = bank.for_call(**meta)
            return _FakeAnthropicResponse(resp)

    class _FakeClient:
        def __init__(self, *a, **kw):
            self.messages = _FakeMessages()

    monkeypatch.setattr(anthropic, "AsyncAnthropic", _FakeClient)
    return bank


def _extract_meta(kwargs: dict) -> dict[str, str]:
    """Extract (agent, company, field) from Anthropic call kwargs.

    Convention: seeder/validator system prompts begin with
        `# META: agent=<name> company=<slug> field=<name>`
    """
    system = kwargs.get("system") or ""
    if isinstance(system, list):
        system = " ".join(b.get("text", "") for b in system if isinstance(b, dict))
    msgs = kwargs.get("messages") or []
    haystack = system + " " + " ".join(
        m.get("content", "") if isinstance(m.get("content"), str) else "" for m in msgs
    )
    parts = {}
    for tag in ("agent", "company", "field"):
        marker = f"{tag}="
        if marker in haystack:
            tail = haystack.split(marker, 1)[1]
            parts[tag] = tail.split()[0].strip(",.;")
    if set(parts) != {"agent", "company", "field"}:
        raise AssertionError(
            f"Could not extract meta from Claude call. Tests must call agents with "
            f"`# META: agent=... company=... field=...` in the system prompt. Got: {parts!r}"
        )
    return parts
```

```python
"""
tests/scripts/record_fixtures.py
─────────────────────────────────
One-shot recorder. Runs the relevant agent against real Claude (requires
LIVE_CLAUDE=1 and ANTHROPIC_API_KEY) and writes JSON to
tests/fixtures/claude_responses/<agent>__<company>__<field>.json.

Usage:
    LIVE_CLAUDE=1 python tests/scripts/record_fixtures.py \
        --agent seeder --company uber --field legal_name --strategy web_search

Re-run after seeder/validator prompt changes to refresh fixtures.
"""
from __future__ import annotations

import argparse
import asyncio
import json
import os
import sys
from pathlib import Path

FIXTURES_DIR = Path(__file__).parent.parent / "fixtures"
RESP_DIR = FIXTURES_DIR / "claude_responses"


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--agent", required=True, choices=["seeder", "validator_a", "validator_b"])
    parser.add_argument("--company", required=True)
    parser.add_argument("--field", required=True)
    parser.add_argument(
        "--strategy",
        default="web_search",
        choices=["web_search", "site_scrape", "filings", "llm_inference"],
    )
    args = parser.parse_args()

    if os.getenv("LIVE_CLAUDE") != "1":
        print("ERROR: set LIVE_CLAUDE=1 to record fixtures.", file=sys.stderr)
        return 2

    asyncio.run(_record(args))
    return 0


async def _record(args) -> None:
    companies = json.loads((FIXTURES_DIR / "test_companies.json").read_text())
    company = companies[args.company]
    payload = await _invoke_agent(args.agent, args.strategy, company, args.field)
    out = RESP_DIR / f"{args.agent}__{args.company}__{args.field}.json"
    out.write_text(json.dumps(payload, indent=2, sort_keys=True))
    print(f"Wrote {out}")


async def _invoke_agent(agent: str, strategy: str, company: dict, field: str) -> dict:
    if agent == "seeder":
        from app.seeding.seeder_agent import SeederAgent
        from app.seeding.field_specs import FIELD_SPECS

        spec = next(
            (s for entity_fields in FIELD_SPECS.values() for s in entity_fields if s.name == field),
            None,
        )
        if spec is None:
            raise SystemExit(f"No FieldSpec for {field!r}")
        result = await SeederAgent.run(spec, strategy, company)
        return {
            "value": result.value,
            "confidence": result.confidence,
            "source_urls": result.source_urls,
            "strategy_used": strategy,
        }
    elif agent.startswith("validator_"):
        from app.validation import validator_a, validator_b

        mod = validator_a if agent == "validator_a" else validator_b
        result = await mod.validate_field(
            company_name=company["company_name"],
            entity_type="org_profiles",
            field_name=field,
            field_label=field.replace("_", " ").title(),
            seeded_value=company["expected"].get(field, ""),
            source_urls=[],
        )
        return result.model_dump() if hasattr(result, "model_dump") else result.__dict__

    raise SystemExit(f"Unknown agent {agent!r}")


if __name__ == "__main__":
    sys.exit(main())
```

```python
    "field": "",

    "primary_source_url": "https://www.example.com/source",
    "notes": "Synthesized fixture from golden values.",
    "confidence": 0.96,
    "duration_ms": 1200,
}


def emit(agent: str, company: str, field: str, value):
    if agent == "seeder":
        payload = {**SEEDER_TEMPLATE, "value": value}
    elif agent == "validator_a":
        payload = {**VALIDATOR_A_TEMPLATE, "field": field, "seeded_value": value}
    else:
        raise SystemExit(agent)
    out = OUT / f"{agent}__{company}__{field}.json"
    out.write_text(json.dumps(payload, indent=2, sort_keys=True))

    for company, sections in GOLDEN.items():

            for field, value in fields.items():

                    continue  # assertion-only, not seeded directly
                emit("seeder", company, field, value)
                emit("validator_a", company, field, value)
    print(f"Wrote fixtures to {OUT}")


if __name__ == "__main__":
    main()
```

```python
"""
app/api/routes/test_route.py
────────────────────────────
Test-only endpoints. Mounted at /api/v1/test ONLY when AEGIS_ENABLE_TEST_ENDPOINTS=1.
Used by Playwright e2e tests. DO NOT MOUNT IN PRODUCTION.
"""
from __future__ import annotations

import json
import os
from pathlib import Path
from uuid import UUID, uuid4

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.auth import create_access_token
from app.database import get_db
from app.models import Organization, User

router = APIRouter(prefix="/test", tags=["test-only"])


class ProvisionRequest(BaseModel):
    fixture_company: str   # "uber" | "stripe" | ...
    role: str = "admin"


class ProvisionResponse(BaseModel):
    email: str
    password: str
    role: str
    orgId: str
    accessToken: str


def _check_enabled():
    if os.getenv("AEGIS_ENABLE_TEST_ENDPOINTS") != "1":
        raise HTTPException(404, "Not found")


@router.post("/provision-tenant", response_model=ProvisionResponse)
async def provision_tenant(payload: ProvisionRequest, db: AsyncSession = Depends(get_db)):
    _check_enabled()
    fixtures_path = Path(__file__).parent.parent.parent.parent / "tests" / "fixtures" / "test_companies.json"
    fixtures = json.loads(fixtures_path.read_text())
    if payload.fixture_company not in fixtures:
        raise HTTPException(400, f"Unknown fixture: {payload.fixture_company}")
    company = fixtures[payload.fixture_company]

    org = Organization(
        name=company["company_name"],
        slug=f"e2e-{uuid4().hex[:8]}",
        onboarding_complete=True,
        fingerprint_data={"company_name": company["company_name"], "domain": company["domain"]},
        is_synthetic=True,
    )
    db.add(org)
    await db.flush()

    user = User(
        email=f"admin+{org.slug}@example.test",
        org_id=org.id,
        role=payload.role,
        hashed_password="$2b$12$abcdefghijklmnopqrstuvwxyz0123456789",
        is_active=True,
    )
    db.add(user)
    await db.commit()

    token = create_access_token({"sub": str(user.id), "org_id": str(org.id)})

    # Run the same seed+validate flow used by real onboarding
    from app.api.routes.orgs_route import _seed_org_background
    await _seed_org_background(
        org_id=str(org.id),
        fingerprint=org.fingerprint_data,
        frameworks=[],
        company_name=company["company_name"],
    )

    return ProvisionResponse(
        email=user.email,
        password="e2e-only",
        role=user.role,
        orgId=str(org.id),
        accessToken=token,
    )


@router.delete("/teardown-tenant/{org_id}")
async def teardown_tenant(org_id: UUID, db: AsyncSession = Depends(get_db)):
    _check_enabled()
    from sqlalchemy import delete
    org = await db.get(Organization, org_id)
    if org and getattr(org, "is_synthetic", False):
        await db.execute(delete(Organization).where(Organization.id == org_id))
        await db.commit()
    return {"deleted": str(org_id)}
```
